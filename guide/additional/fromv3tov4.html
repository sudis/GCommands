# Updating from v3 to v4

Dev build now

+ cmd.userRequiredRole

- return (in slash cmd)
+ respond()
+ edit()

+ new MessageButton().setStyle("gray").setLabel("test").setID("custom_id").toJSON()





 


 




 


 




const { MessageButton } = require("gcommands")

module.exports = {
	name: "test",
	description: "Test",
	run: async({client, interaction, member, message, guild, channel, respond, edit}, args) => {
		const button = new MessageButton().setStyle("gray").setLabel("test").setID("custom_id").toJSON()

		respond({
			content: "My ping is `" + Math.round(client.ws.ping) + "ms`",
			ephemeral: true,
			allowedMentions: { parse: [], repliedUser: true },
			components: [button]
		})

		setTimeout(() => {
			edit("hi")
		}, 3000)
	}
};
